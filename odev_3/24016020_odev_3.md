        if int(girdi)<0:
            return "sayi negatif."

class Ucus():

    def __init__(self, kod, kalkis, varis, sure, kapasite, yolcu): #attribute atamasi, self kendisi demek ve fonksiyona tanimli dolayisiyla ondan sonrakiler attribute olarak tanimlaniyor.

        self.kalkis = kalkis
        self.varis = varis
